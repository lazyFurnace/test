module.exports = React.createClass({
  onClick:function(index){
  	var callBack = this.props.customHandler;
    if(callBack){
      callBack({
      	data:index,
        eventType:"click"
      });
    }
  },
  render: function(){
    
    var thead = [];
    this.props.customData.title.map(function(d){
    	thead.push(<td>{d}</td>);
    })
    var tbody = [];
    this.props.customData.content.map(function(d,i){
    	tbody.push(<TbodyList dataClick={this.onClick} listIndex={i} listData={d} />)
    }.bind(this));
    return (
      <table>
      	<thead>
        	<tr>{thead}</tr>
        </thead>
        <tbody>
          {tbody}
        </tbody>
      </table>
    )
  }
});
var TbodyList = React.createClass({
  handleClick:function(e){
    this.props.dataClick(e.target.title);
  },
	render:function(){
    var list = this.props.listData;
  	return (
    	<tr>
      	<td onClick={this.handleClick} title={this.props.listIndex}>{list[0]}</td>
        <td>{list[1]}</td>
        <td>{list[2]}</td>
      </tr>
    )
  }
})
